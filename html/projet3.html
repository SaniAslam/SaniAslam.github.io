"""Fonction f1 :
Écrire une fonction nommée f1 qui possède comme seul argument une liste de nombres entiers et
qui retourne la somme de tous les éléments de cette liste."""

def f1(l1: list) -> int:
    total = 0
    for i in l1:
        total += i
    return total

print(f1([4, 5, 6, 2, 8]))


"""Fonction f2 :
Écrire une fonction nommée f2 qui prend en argument une liste de chaînes de caractères, et qui
retourne une chaîne contenant tous les éléments de la liste concaténés, séparés par un slash ("/")."""

def f2(liste: list) -> str:
    result = ""
    for i in liste:
        result += i + "/"
    return result

print(f2(["chat", "chien", "oiseau"]))


"""Fonction f3 :
Écrire une fonction nommée f3 qui prend en argument une liste de nombres entiers et un entier x.
La fonction doit retourner une nouvelle liste où chaque élément de la liste d'origine est multiplié par x"""

def f3(liste: list, x: int) -> list:
    result = []
    for i in liste:
        result.append(i * x)
    return result

print(f3([2, 1, 5, 10], 2))


"""Fonction f4 :
Écrire une fonction nommée f4 qui prend en argument un tuple de nombres entiers, et qui retourne
la valeur maximale parmi les éléments de ce tuple."""

def f4(a: tuple) -> int:
    max = 0
    for i in a:
        if max < i:
            max = i
    return max

print(f4((5, 7, 9, 6, 3, 4)))


"""Fonction f5 :
Écrire une fonction nommée f5 qui prend en argument une chaîne de caractères, et qui retourne la
même chaîne mais inversée."""

def f5(a: str) -> str:
    reverse = ""
    for i in a:
        reverse = i + reverse
    return reverse

print(f5("Azerty"))


"""Fonction f6 :
Écrire une fonction nommée f6 qui prend en argument un entier n, et qui retourne la somme des
entiers de 0 à n-1."""

def f6(a: int) -> int:
    result = 0
    for i in range(a):
        result += i
    return result

print(f6(5))


"""Fonction f7 :
Écrire une fonction nommée f7 qui prend en argument une liste de nombres entiers, et qui retourne
une nouvelle liste contenant uniquement les nombres pairs présents dans la liste d'origine."""

def f7(a: list) -> list:
    pair = []
    for i in a:
        if i % 2 == 0:
            pair.append(i)
    return pair

print(f7([4, 0, 6, 8, 45, 692, 15, 75, 15]))


"""Fonction f8 :
Écrire une fonction nommée f8 qui prend en argument une chaîne de caractères, et qui retourne le
nombre de voyelles (a, e, i, o, u, y) présentes dans cette chaîne."""

def f8(a: str) -> int:
    nombre = 0
    voyelles = ["a", "e", "i", "o", "u", "y", "A", "E", "I", "O", "U", "Y"]
    for i in a:
        if i in voyelles:
            nombre += 1
    return nombre

print(f8("Martin, Lucien, Alexandre"))


"""Fonction f9 :
Écrire une fonction nommée f9 qui prend en argument une liste de nombres entiers, et qui retourne
une nouvelle liste contenant les carrés des éléments de la liste d'origine"""

def f9(liste: list) -> list:
    result = []
    for i in liste:
        i = i ** 2
        result.append(i)
    return result

print(f9([5, 8, 9, 4, 1]))


"""Fonction f10 :
Écrire une fonction nommée f10 qui prend en arguments deux entiers a et b, et qui retourne un
tuple contenant tous les entiers entre a et b-1 qui sont divisibles par 3."""

def f10(a: int, b: int) -> tuple:
    result = []
    for i in range(a, b):
        if i % 3 == 0:
            result.append(i)
    return result

print(f10(2, 15))
